def calculate_zero_crossing(x):
    tmp = np.nonzero(np.diff(np.signbit(x)))[0]
    return len(tmp)

def calculate_entropy(x):
    hist, _ = np.histogram(x, bins=50, density=True)
    probabilities = hist / hist.sum()
    return entropy(probabilities)

    ecg = pd.to_numeric(ecg, errors='coerce')

    coef = pywt.wavedec(ecg, wavelet='db4', level=4)
